-- inner join
create table customers(cid int,name varchar(20));

insert into customers values(1,'ABC');
insert into customers values(2,'PQR');
insert into customers values(3,'XYZ');

create table orders(oid int,oname varchar(20),cid int,amount int);

insert into orders values(101,'APPLE',1,200);
insert into orders values(102,'Orange',2,300);
insert into orders values(103,'Pizza',5,500);

select customers.cid,customers.name,orders.oid,orders.amount from customers inner join orders on customers.cid = orders.cid;

-- abbreviating table name to 'c' and 'o' for convenience
select c.cid,c.name,o.oid,o.amount from customers c inner join orders o
on c.cid = o.cid;

-- left outer join (all the records from left table w null for non-matching rows in right table)
select c.cid,c.name,o.oid,o.amount from customers c left outer join orders o
on c.cid = o.cid;

-- right outer join (all the records from right table w null for non-matching rows in left table)
select c.cid,c.name,o.oid,o.amount from customers c right outer join orders o
on c.cid = o.cid;

-- full outer join (all the records from both tables w null for non-matching rows)
select c.cid,c.name,o.oid,o.amount from customers c full outer join orders o
on c.cid = o.cid;

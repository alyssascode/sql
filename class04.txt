-- CHECK (limit input)
drop table demo;
create table demo (id int primary key, name varchar(20), age int, check (age > 18));
insert into demo (id, name, age) values (1, 'ABC', 34);
insert into demo (id, name, age) values (1, 'PQR', 18);
select * from demo;

-- DEFAULT (sets a column default -- have to pass values in 1st paren clause
